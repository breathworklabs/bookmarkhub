## 🧵 **Full Thread Capture Implementation**

### **Desktop Browser Extension - Complete Thread Capture**

The extension can capture entire threads automatically when someone bookmarks any tweet in the thread, without needing Twitter API access.

```javascript
// content-script.js - Injected into Twitter/X pages
class SmartThreadCapture {
  async captureOnBookmark() {
    // Listen for bookmark button clicks
    document.addEventListener('click', async (e) => {
      const target = e.target as HTMLElement;

      // Check if bookmark button was clicked
      if (this.isBookmarkButton(target)) {
        const tweetElement = target.closest('article');

        // Start capture immediately
        const thread = await this.intelligentCapture(tweetElement);

        // Store locally first (instant)
        await this.storeLocally(thread);

        // Sync to cloud (background)
        this.syncToCloud(thread);

        // Show user feedback
        this.showNotification('Thread captured! ' + thread.tweets.length + ' tweets saved');
      }
    });
  }

  async intelligentCapture(bookmarkedTweet: HTMLElement) {
    const strategy = this.detectThreadStrategy(bookmarkedTweet);

    switch(strategy) {
      case 'FULL_THREAD_VISIBLE':
        // All tweets are already loaded
        return this.captureVisibleThread(bookmarkedTweet);

      case 'PARTIAL_THREAD':
        // Need to expand "Show this thread"
        return this.captureExpandedThread(bookmarkedTweet);

      case 'QUOTE_TWEET':
        // Capture both the quote and quoted tweet
        return this.captureQuoteTweet(bookmarkedTweet);

      case 'REPLY_CHAIN':
        // Capture reply context
        return this.captureReplyChain(bookmarkedTweet);

      default:
        // Single tweet
        return this.captureSingleTweet(bookmarkedTweet);
    }
  }

  async captureVisibleThread(startTweet: HTMLElement) {
    const tweets: Tweet[] = [];
    const container = startTweet.closest('[data-testid="primaryColumn"]');

    // Find all connected tweets
    const articles = container?.querySelectorAll('article') || [];

    let inThread = false;
    let threadAuthor = this.getAuthor(startTweet);

    for (const article of articles) {
      const author = this.getAuthor(article as HTMLElement);

      // Start capturing when we find the thread
      if (!inThread && article === startTweet) {
        inThread = true;
      }

      // Continue capturing same-author tweets
      if (inThread && author === threadAuthor) {
        tweets.push(this.extractTweetData(article as HTMLElement));
      } else if (inThread && author !== threadAuthor) {
        // Thread ended
        break;
      }
    }

    return { tweets, complete: true };
  }

  extractTweetData(element: HTMLElement): Tweet {
    return {
      id: this.getTweetId(element),
      text: element.querySelector('[data-testid="tweetText"]')?.textContent || '',
      author: {
        username: element.querySelector('[href^="/"]')?.textContent?.replace('@', '') || '',
        displayName: element.querySelector('[data-testid="User-Name"]')?.firstChild?.textContent || '',
        avatarUrl: element.querySelector('img[src*="profile_images"]')?.getAttribute('src') || ''
      },
      media: this.extractAllMedia(element),
      metrics: {
        replies: this.parseMetric(element, 'reply'),
        retweets: this.parseMetric(element, 'retweet'),
        likes: this.parseMetric(element, 'like'),
        bookmarks: this.parseMetric(element, 'bookmark')
      },
      timestamp: this.getTimestamp(element),
      html: element.outerHTML,
      preserved: true
    };
  }

  extractAllMedia(element: HTMLElement) {
    const media = {
      images: [] as Array<{url: string, alt: string}>,
      videos: [] as Array<{thumbnail: string, duration?: string}>,
      links: [] as Array<{url: string, display: string}>,
      card: null as any
    };

    // Images
    element.querySelectorAll('img[src*="/media/"]').forEach(img => {
      media.images.push({
        url: (img as HTMLImageElement).src.replace(/&name=\w+$/, '&name=large'),
        alt: img.getAttribute('alt') || ''
      });
    });

    // Videos
    element.querySelectorAll('video').forEach(video => {
      media.videos.push({
        thumbnail: video.getAttribute('poster') || '',
        duration: video.getAttribute('data-duration')
      });
    });

    // Link cards
    const card = element.querySelector('[data-testid="card.wrapper"]');
    if (card) {
      media.card = {
        url: card.querySelector('a')?.href,
        title: card.querySelector('[dir="ltr"]')?.textContent,
        description: card.querySelector('[data-testid="card.layoutSmall.detail"]')?.textContent,
        image: card.querySelector('img')?.src
      };
    }

    return media;
  }

  async findThreadBoundaries(tweetElement: HTMLElement) {
    const thread = {
      tweets: [],
      bookmarkedIndex: -1,
      authorUsername: this.extractUsername(tweetElement)
    };

    // Traverse UP to find thread start
    let currentElement = tweetElement;
    const upwardTweets = [];

    while (currentElement) {
      const prevTweet = this.getPreviousTweet(currentElement);

      // Check if tweets are connected (same author + visual thread line)
      if (this.isPartOfThread(prevTweet, currentElement)) {
        upwardTweets.unshift(this.extractTweetData(prevTweet));
        currentElement = prevTweet;
      } else {
        break;
      }
    }

    // Add bookmarked tweet
    const bookmarkedData = this.extractTweetData(tweetElement);

    // Traverse DOWN to find thread end
    currentElement = tweetElement;
    const downwardTweets = [];

    while (currentElement) {
      const nextTweet = this.getNextTweet(currentElement);

      if (this.isPartOfThread(currentElement, nextTweet)) {
        downwardTweets.push(this.extractTweetData(nextTweet));
        currentElement = nextTweet;
      } else {
        break;
      }
    }

    // Combine all tweets
    thread.tweets = [...upwardTweets, bookmarkedData, ...downwardTweets];
    thread.bookmarkedIndex = upwardTweets.length;

    return thread;
  }

  isPartOfThread(tweet1: HTMLElement, tweet2: HTMLElement): boolean {
    if (!tweet1 || !tweet2) return false;

    // Check if tweets are from same author
    const author1 = tweet1?.querySelector('[data-testid="User-Name"]')?.innerText;
    const author2 = tweet2?.querySelector('[data-testid="User-Name"]')?.innerText;

    if (author1 !== author2) return false;

    // Check for visual thread connector (the vertical line)
    const hasThreadLine = tweet2?.querySelector('[style*="bottom: 0px"]');

    // Check timestamp proximity (threads usually posted close together)
    const time1 = this.extractTimestamp(tweet1);
    const time2 = this.extractTimestamp(tweet2);
    const timeDiff = Math.abs(time2 - time1);

    return hasThreadLine || timeDiff < 3600000; // Within 1 hour
  }
}

// Monitor for bookmark actions
const observer = new MutationObserver(async (mutations) => {
  for (const mutation of mutations) {
    if (mutation.type === 'attributes' &&
        mutation.attributeName === 'data-testid' &&
        mutation.target.getAttribute('data-testid')?.includes('bookmark')) {

      // User just bookmarked - capture the thread!
      const tweetElement = mutation.target.closest('article');
      const threadCapture = new SmartThreadCapture();
      const fullThread = await threadCapture.captureFullThread(tweetElement);

      // Send to background script for storage
      chrome.runtime.sendMessage({
        type: 'THREAD_CAPTURED',
        data: fullThread
      });
    }
  }
});

// Start observing
observer.observe(document.body, {
  attributes: true,
  subtree: true,
  attributeFilter: ['data-testid']
});
```

## 📱 **Share Sheet Integration - Best Mobile Solution**

### **What is Share Sheet Integration?**

Share Sheet Integration is a **native mobile feature** that allows BookmarksX to receive content shared from the Twitter/X app. When users tap the "Share" button in Twitter, BookmarksX appears as an option in the native share menu.

```

User selects BookmarksX → Tweet URL sent to your app →
```

```swift
class ShareViewController: SLComposeServiceViewController {
    override func isContentValid() -> Bool {
        // Validate that we have a Twitter URL
        return hasTwitterURL()
    }

    override func didSelectPost() {
        // Extract the Twitter URL from shared content
        if let item = extensionContext?.inputItems.first as? NSExtensionItem,
           let provider = item.attachments?.first {

            provider.loadItem(forTypeIdentifier: "public.url") { (url, error) in
                if let tweetURL = url as? URL {
                    // Send to your app for processing
                    self.saveToBookmarksX(tweetURL)
                }
            }
        }

        // Close the share sheet
        self.extensionContext?.completeRequest(returningItems: [], completionHandler: nil)

        // Option 1: Save to shared app group (offline)

        savedURLs.append(url.absoluteString)
        sharedDefaults?.set(savedURLs, forKey: "pendingBookmarks")

        // Option 2: Send to your server (online)
        let request = URLRequest(url: URL(string: "https://api.bookmarksx.app/capture")!)
        // ... send tweet URL to your backend
    }
}
```

### **Android Implementation**

```kotlin
// Android - ShareActivity.kt
class ShareActivity : Activity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        when (intent?.action) {
            Intent.ACTION_SEND -> {
                if ("text/plain" == intent.type) {
                    handleSharedText(intent)
                }
            }
        }
    }

    private fun handleSharedText(intent: Intent) {
        intent.getStringExtra(Intent.EXTRA_TEXT)?.let { sharedText ->
            // Extract Twitter URL from shared text
            val tweetUrl = extractTwitterUrl(sharedText)

            // Save to your app
            tweetUrl?.let {
                saveBookmark(it)
                Toast.makeText(this, "Tweet saved to BookmarksX!", Toast.LENGTH_SHORT).show()
            }

            // Close the activity
            finish()
        }
    }

    private fun extractTwitterUrl(text: String): String? {
        val pattern = "https?://(twitter\\.com|x\\.com)/\\w+/status/\\d+"
        val regex = Regex(pattern)
        return regex.find(text)?.value
    }

    private fun saveBookmark(url: String) {
        // Save to local database
        val bookmark = Bookmark(
            url = url,
            timestamp = System.currentTimeMillis(),
            status = "pending_capture"
        )

        // Queue for background processing
        WorkManager.getInstance(this).enqueue(
            OneTimeWorkRequestBuilder<CaptureWorker>()
                .setInputData(workDataOf("tweet_url" to url))
                .build()
        )
    }
}
```

### **Backend Processing**

```typescript
// API endpoint to receive shared tweets from mobile
app.post('/api/capture/mobile', async (req, res) => {
  const { tweetUrl, userId, deviceId } = req.body;

  try {
    // Parse tweet ID from URL
    const tweetId = extractTweetId(tweetUrl);

    // Fetch tweet content (using puppeteer or similar)
    const tweetData = await captureTweetContent(tweetUrl);

    // Check if it's part of a thread
    if (tweetData.isThread) {
      const fullThread = await captureFullThread(tweetUrl);
      tweetData.thread = fullThread;
    }

    // Store in database
    await db.bookmarks.create({
      userId,
      tweetId,
      url: tweetUrl,
      content: tweetData,
      source: 'mobile_share',
      deviceId,
      capturedAt: new Date()
    });

    // Return success with thread info
    res.json({
      success: true,
      message: `Saved ${tweetData.thread?.length || 1} tweets`,
      bookmarkId: tweetId
    });

  } catch (error) {
    console.error('Capture failed:', error);
    res.status(500).json({ error: 'Failed to capture tweet' });
  }
});
```
